--2
select nommat, avg (calif) "Promedio"
from mater m, historial h
where calif>=6 and extract (year from fecha) = extract(year from sysdate)-1 and m.clavem=h.clavem
group by nommat

--3
select nommat, count(distinct calif)
from mater m, historial h
where m.clavem=h.clavem
group by nommat
having count(distinct calif)>2

---Escribir el nombre de las personas egresadas del ITAM que ganaron algún lugar en algún
--concurso. Escribir también el nombre de la carrera que estudiaron.
select  nomA, nomorg,  nomcar
from organización org, autor a , estudió e, carrera c
where nomorg ='ITAM' and org.idorg=e.idorg and a.ida =e.ida and e.idcar=c.idcar

--Escribir el nombre de las organizaciones que no organizaron concurso alguno el año pasado.
--Ordenar alfabéticamente

select nomorg
from organización  org, ORGANIZÓ o, concurso c
